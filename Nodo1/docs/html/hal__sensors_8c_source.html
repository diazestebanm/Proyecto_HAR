141static void qmi_calibrate_bias(void)
142{
143 const int N = 50;
144 int64_t sx = 0, sy = 0, sz = 0;

146 int16_t ax, ay, az, gx, gy, gz;

149 sy += gy;

154 s_gx_bias = (int32_t)(sx / N);
155 s_gy_bias = (int32_t)(sy / N);
156 s_gz_bias = (int32_t)(sz / N);
157 s_imu_bias_ok = true;

216static int ppg_copy(uint16_t *r, uint16_t *ir)
217{
218 int n = s_ppg_count;
219 int start = s_ppg_head - n;
220 if (start < 0) start += APP_PPG_WINDOW_SAMPLES;
221 for (int i = 0; i < n; ++i) {
222 int idx = (start + i) % APP_PPG_WINDOW_SAMPLES;
223 r[i] = s_red_buf[idx];
224 ir[i] = s_ir_buf[idx];
225 }
226 return n;
227}

240static void ppg_analyze(ppg_metrics_t *out)
241{
242 static uint16_t r[APP_PPG_WINDOW_SAMPLES];
243 static uint16_t ir[APP_PPG_WINDOW_SAMPLES];
244 static float ir_ac[APP_PPG_WINDOW_SAMPLES];
245
246 memset(out, 0, sizeof(*out));
247 int n = ppg_copy(r, ir);
248 if (n < 200) return;
249
250 double sum_r = 0.0, sum_ir = 0.0;
251 uint16_t min_ir = 0xFFFF, max_ir = 0;
252 for (int i = 0; i < n; ++i) {
253 sum_r += r[i];
254 sum_ir += ir[i];
255 if (ir[i] < min_ir) min_ir = ir[i];
256 if (ir[i] > max_ir) max_ir = ir[i];
257 }
258
259 float dc_r = (float)(sum_r / n);
260 float dc_ir = (float)(sum_ir / n);
261 out->dc_ir = (uint16_t)dc_ir;
262 out->finger_present = (dc_ir > 8000.0f) && ((max_ir - min_ir) > 50);
263
264 if (!out->finger_present) return;
265
266 double sq_r = 0.0, sq_ir = 0.0;
267 for (int i = 0; i < n; ++i) {
268 float rac = (float)r[i] - dc_r;
269 float iac = (float)ir[i] - dc_ir;
270 sq_r += rac * rac;
271 sq_ir += iac * iac;
272 if (i == 0 || i == n - 1) ir_ac[i] = iac;
273 else ir_ac[i] = (((float)ir[i - 1] + ir[i] + ir[i + 1]) / 3.0f) - dc_ir;
274 }
275
276 float ac_r = sqrtf((float)(sq_r / n));
277 float ac_ir = sqrtf((float)(sq_ir / n));
278
279 if (dc_r > 1 && dc_ir > 1 && ac_r > 1 && ac_ir > 1) {
280 float ratio = (ac_r / dc_r) / (ac_ir / dc_ir);
281 float spo2 = 110.0f - 25.0f * ratio;
282 if (spo2 >= 70.0f && spo2 <= 100.0f) {
283 out->spo2 = clamp_u8(spo2, 0, 100);
284 out->spo2_valid = true;
285 }
286 }
287
288 float thr = fmaxf(25.0f, ac_ir * 0.5f);
289 int last_peak = -1000;
290 int intervals = 0;
291 int interval_sum = 0;
292
293 for (int i = 1; i < n - 1; ++i) {
294 bool peak = (ir_ac[i] > thr) && (ir_ac[i] > ir_ac[i - 1]) && (ir_ac[i] >= ir_ac[i + 1]);
295 if (peak && (i - last_peak) >= 35) {
296 if (last_peak >= 0) {
297 interval_sum += (i - last_peak);
298 intervals++;
299 }
300 last_peak = i;
301 }
302 }
303
304 if (intervals > 0) {
305 float avg = (float)interval_sum / intervals;
306 float bpm = 60.0f * APP_PPG_SAMPLE_RATE_HZ / avg;
307 if (bpm >= 40.0f && bpm <= 200.0f) {
308 out->bpm = clamp_u8(bpm, 0, 255);
309 out->bpm_valid = true;
310 }
311 }
312}

332static float vbat_to_percent(float v)
333{
334 static const float map_v[] = {4.20f,4.10f,4.00f,3.90f,3.80f,3.70f,3.60f,3.50f,3.40f,3.30f,3.20f,3.10f,3.00f};
335 static const float map_pct[] = {100.f,95.f,88.f,78.f,65.f,50.f,36.f,24.f,14.f,8.f,4.f,1.f,0.f};
336 const int N = 13;
337
338 if (v >= map_v[0]) return 100.0f;
339 if (v <= map_v[N - 1]) return 0.0f;
340
341 for (int i = 0; i < N - 1; ++i) {
342 if (v >= map_v[i + 1]) {
343 float t = (v - map_v[i + 1]) / (map_v[i] - map_v[i + 1]);
344 return map_pct[i + 1] + t * (map_pct[i] - map_pct[i + 1]);
345 }
346 }
347 return 0.0f;
348}

510esp_err_t hal_battery_read(battery_status_t *out)
511{
512 int samples[32];
513 for (int i = 0; i < 32; ++i) {
514 ESP_RETURN_ON_ERROR(adc_oneshot_read(s_adc_handle, APP_BAT_ADC_CHANNEL, &samples[i]), TAG, "ADC read");
515 vTaskDelay(pdMS_TO_TICKS(4));
516 }
517
518 qsort(samples, 32, sizeof(int), cmp_int);
519 int raw = (samples[15] + samples[16]) / 2;
520 int mv_gpio = raw_to_mv(raw);
521 float vbat = (mv_gpio / 1000.0f) * APP_BAT_DIV_RATIO;
522
523 if (s_ema_vbat < 0.0f) s_ema_vbat = vbat;
524 else s_ema_vbat = 0.15f * vbat + 0.85f * s_ema_vbat;
525
526 float pct = vbat_to_percent(s_ema_vbat);
527 if (s_last_pct < 0.0f || fabsf(pct - s_last_pct) >= 2.0f) s_last_pct = pct;
528
529 out->mv = (uint16_t)(s_ema_vbat * 1000.0f);
530 out->pct = clamp_u8(s_last_pct, 0, 100);
531 out->valid = (out->mv >= 2800 && out->mv <= 4400);
532 out->critical = out->valid && (out->mv <= APP_BAT_CRITICAL_MV);
533 out->low = out->valid && (out->mv <= APP_BAT_LOW_MV);
534
535 return ESP_OK;
536}
