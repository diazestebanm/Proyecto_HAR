92esp_err_t app_radio_send_blocking(const adl_payload_t *payload, bool critical)
93{
94 const int tries = critical ? 2 : 1;
95 s_sender_task = xTaskGetCurrentTaskHandle();
96
97 for (int i = 0; i < tries; ++i) {

99 uint32_t ok = ulTaskNotifyTake(pdTRUE, pdMS_TO_TICKS(APP_RADIO_SEND_TIMEOUT_MS));
100 if (ok && s_last_status == ESP_NOW_SEND_SUCCESS) {
101 return ESP_OK;
