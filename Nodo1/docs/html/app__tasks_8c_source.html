62static uint16_t build_flags(const imu_features_t *imu, const ppg_metrics_t *ppg, const battery_status_t *bat, bool boot_event)
63{
64 uint16_t f = 0;
65
66 if (boot_event) f |= APP_FLAG_BOOT_EVENT;
67 if (imu->valid) f |= APP_FLAG_IMU_VALID_WINDOW;
68 if (imu->sample_count >= APP_IMU_EXPECTED_SAMPLES) f |= APP_FLAG_IMU_WINDOW_COMPLETE;
69 if (imu->sample_count < APP_IMU_EXPECTED_SAMPLES) f |= APP_FLAG_IMU_SAMPLE_LOSS;
70 if (hal_imu_bias_ok()) f |= APP_FLAG_GYRO_BIAS_OK;
71
72 if (ppg->finger_present) f |= APP_FLAG_PPG_FINGER_PRESENT;
73 if (ppg->bpm_valid) f |= APP_FLAG_BPM_VALID;
74 if (ppg->spo2_valid) f |= APP_FLAG_SPO2_VALID;
75
76 if (bat->valid) f |= APP_FLAG_BAT_VALID;
77 if (bat->low) f |= APP_FLAG_BAT_LOW;
78 if (bat->critical) f |= APP_FLAG_BAT_CRITICAL;
79
80 if (app_radio_link_degraded()) f |= APP_FLAG_LINK_DEGRADED;
81 return f;
82}

113static void task_imu(void *arg)
114{
115 app_signal_ctx_t sig;
116 imu_sample_t s;
117
118 while (1) {
119 uint32_t n = ulTaskNotifyTake(pdTRUE, portMAX_DELAY);
120 if (!n) continue;
121
122 app_signal_reset(&sig);
123 hal_imu_start();
124
125 TickType_t last = xTaskGetTickCount();
126 for (uint32_t i = 0; i < APP_IMU_EXPECTED_SAMPLES; ++i) {
127 if (hal_imu_read_sample(&s) == ESP_OK) {
128 s.t_ms = rel_ms();
129 app_signal_push_imu(&sig, &s);
130 }
131 vTaskDelayUntil(&last, pdMS_TO_TICKS(APP_IMU_SAMPLE_PERIOD_MS));
132 }
133
134 hal_imu_stop();
135
136 imu_features_t feat;
137 app_signal_build_features(&sig, &feat);
138
139 xSemaphoreTake(s_data_mutex, portMAX_DELAY);
140 s_last_imu = feat;
141 xSemaphoreGive(s_data_mutex);
142
143 xTaskNotify(s_task_ctrl, NTF_IMU_DONE, eSetBits);
144 }
145}

168static void enqueue_payload(app_msg_type_t type, bool critical, bool boot_event)
169{
170 ppg_metrics_t ppg;
171 battery_status_t bat;
172 imu_features_t imu;
173
174 xSemaphoreTake(s_data_mutex, portMAX_DELAY);
175 ppg = s_last_ppg;
176 bat = s_last_bat;
177 imu = s_last_imu;
178 xSemaphoreGive(s_data_mutex);
179
180 uint16_t flags = build_flags(&imu, &ppg, &bat, boot_event);
181
182 adl_payload_t payload;
183 app_signal_build_payload(&payload,
184 type,
185 s_boot_id,
186 ++s_seq,
187 rel_ms(),
188 flags,
189 &imu,
190 &ppg,
191 &bat);
192
193 radio_tx_item_t item = {
194 .payload = payload,
195 .critical = critical
196 };
197 xQueueSend(s_radio_queue, &item, portMAX_DELAY);
198}

206static void task_control_fsm(void *arg)
207{
208 app_state_t state = APP_STATE_BOOT;
209 TickType_t last_bat = 0;
210 bool boot_event_pending = true;
211
212 hal_battery_read(&s_last_bat);
213
214 while (1) {
215 switch (state) {
216 case APP_STATE_BOOT:
217 enqueue_payload(APP_MSG_ALARM, true, true);
218 boot_event_pending = false;
219 state = APP_STATE_IDLE_LOW_POWER;
220 break;
221
222 case APP_STATE_IDLE_LOW_POWER:
223 if ((xTaskGetTickCount() - last_bat) >= pdMS_TO_TICKS(APP_BAT_ACTIVE_PERIOD_MS)) {
224 battery_status_t bat;
225 if (hal_battery_read(&bat) == ESP_OK) {
226 xSemaphoreTake(s_data_mutex, portMAX_DELAY);
227 s_last_bat = bat;
228 xSemaphoreGive(s_data_mutex);
229 }
230 last_bat = xTaskGetTickCount();
231 }
232 state = APP_STATE_ACQUIRE_IMU;
233 break;
234
235 case APP_STATE_ACQUIRE_IMU: {
236 xTaskNotifyGive(s_task_imu);
237 uint32_t notified = 0;
238 xTaskNotifyWait(0, UINT32_MAX, &notified, pdMS_TO_TICKS(APP_IMU_WINDOW_MS + 1000));
239 state = APP_STATE_ACQUIRE_PPG;
240 break;
241 }
242
243 case APP_STATE_ACQUIRE_PPG:
244 state = APP_STATE_BUILD_PAYLOAD;
245 break;
246
247 case APP_STATE_BUILD_PAYLOAD:
248 if (s_last_bat.critical || app_radio_link_degraded() || boot_event_pending) {
249 state = APP_STATE_TX_ALARM;
250 } else {
251 state = APP_STATE_TX_NORMAL;
252 }
253 break;
254
255 case APP_STATE_TX_NORMAL:
256 enqueue_payload(APP_MSG_NORMAL, false, false);
257 boot_event_pending = false;
258 vTaskDelay(pdMS_TO_TICKS(APP_TX_PERIOD_MS));
259 state = APP_STATE_IDLE_LOW_POWER;
260 break;
261
262 case APP_STATE_TX_ALARM:
263 enqueue_payload(APP_MSG_ALARM, true, boot_event_pending);
264 boot_event_pending = false;
265 vTaskDelay(pdMS_TO_TICKS(500));
266 state = APP_STATE_RECOVERY;
267 break;
268
269 case APP_STATE_RECOVERY:
270 vTaskDelay(pdMS_TO_TICKS(1000));
271 state = APP_STATE_IDLE_LOW_POWER;
272 break;
273
274 default:
275 state = APP_STATE_RECOVERY;
276 break;
277 }
278 }

285esp_err_t app_tasks_start(void)
286{
287 s_boot_id = (uint16_t)(esp_random() & 0xFFFF);
288 s_t0_us = esp_timer_get_time();
289
290 s_radio_queue = xQueueCreate(8, sizeof(radio_tx_item_t));
291 s_data_mutex = xSemaphoreCreateMutex();
292
293 xTaskCreate(task_ppg, "TaskPPG", 4096, NULL, 6, &s_task_ppg);
294
295 ESP_ERROR_CHECK(hal_sensors_init(s_task_ppg));
296 ESP_ERROR_CHECK(app_radio_init());
297 s_hal_ready = true;
298
299 xTaskCreate(task_imu, "TaskIMU", 4096, NULL, 5, &s_task_imu);
300 xTaskCreate(task_radio, "TaskRadio", 4096, NULL, 4, &s_task_radio);
301 xTaskCreate(task_control_fsm, "TaskControlFSM", 4096, NULL, 7, &s_task_ctrl);
302
303 return ESP_OK;
304}
