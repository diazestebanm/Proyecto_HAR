39void app_signal_push_imu(app_signal_ctx_t *ctx, const imu_sample_t *s)
40{
41 float ax = s->ax * APP_ACC_MG_PER_LSB;
42 float ay = s->ay * APP_ACC_MG_PER_LSB;
43 float az = s->az * APP_ACC_MG_PER_LSB;
44
45 float gx = s->gx * APP_GYRO_DPS10_PER_LSB;
46 float gy = s->gy * APP_GYRO_DPS10_PER_LSB;
47 float gz = s->gz * APP_GYRO_DPS10_PER_LSB;
48
49 float amag = sqrtf(ax * ax + ay * ay + az * az);
50 float gmag = sqrtf(gx * gx + gy * gy + gz * gz);
51
52 ctx->sum_acc[0] += ax; ctx->sum_acc[1] += ay; ctx->sum_acc[2] += az;
53 ctx->sum2_acc[0] += ax * ax; ctx->sum2_acc[1] += ay * ay; ctx->sum2_acc[2] += az * az;
54
55 ctx->sum_gyro[0] += gx; ctx->sum_gyro[1] += gy; ctx->sum_gyro[2] += gz;
56 ctx->sum2_gyro[0] += gx * gx; ctx->sum2_gyro[1] += gy * gy; ctx->sum2_gyro[2] += gz * gz;
57
58 ctx->sum_acc_mag += amag;
59 ctx->sum2_acc_mag += amag * amag;
60 ctx->sum_gyro_mag += gmag;
61 ctx->sum2_gyro_mag += gmag * gmag;
62
63 ctx->sum_sma += fabsf(ax) + fabsf(ay) + fabsf(az);
64
65 if (ctx->count < (APP_IMU_EXPECTED_SAMPLES + 4)) {
66 ctx->acc_mag_hist[ctx->count] = amag;
67 }
68 ctx->count++;
69}

80static float stdf(float sum, float sum2, float n)
81{
82 if (n <= 1) return 0.0f;
83 float mu = sum / n;
84 float v = (sum2 / n) - mu * mu;
85 return (v > 0) ? sqrtf(v) : 0.0f;
86}

93void app_signal_build_features(const app_signal_ctx_t *ctx, imu_features_t *out)
94{
95 memset(out, 0, sizeof(*out));
96 out->sample_count = ctx->count;
97
98 uint16_t min_valid = (APP_IMU_EXPECTED_SAMPLES * APP_IMU_MIN_VALID_PCT) / 100U;
99 out->valid = (ctx->count >= min_valid);
100
101 float n = (float)ctx->count;
102 if (n <= 0) return;
103
104 for (int i = 0; i < 3; ++i) {
105 float am = meanf(ctx->sum_acc[i], n);
106 float as = stdf(ctx->sum_acc[i], ctx->sum2_acc[i], n);
107 float ar = sqrtf(ctx->sum2_acc[i] / n);
108
109 float gm = meanf(ctx->sum_gyro[i], n);
110 float gs = stdf(ctx->sum_gyro[i], ctx->sum2_gyro[i], n);
111 float gr = sqrtf(ctx->sum2_gyro[i] / n);
112
113 out->acc_mean[i] = q16(am);
114 out->acc_std[i] = q16(as);
115 out->acc_rms[i] = q16(ar);
116
117 out->gyro_mean[i] = q16(gm);
118 out->gyro_std[i] = q16(gs);
119 out->gyro_rms[i] = q16(gr);
120 }
121
122 out->acc_mag_mean = q16(meanf(ctx->sum_acc_mag, n));
123 out->acc_mag_std = q16(stdf(ctx->sum_acc_mag, ctx->sum2_acc_mag, n));
124 out->acc_sma = q16(meanf(ctx->sum_sma, n));
125
126 out->gyro_mag_mean = q16(meanf(ctx->sum_gyro_mag, n));
127 out->gyro_mag_std = q16(stdf(ctx->sum_gyro_mag, ctx->sum2_gyro_mag, n));
128
129 uint8_t peaks = 0;
130 float thr = meanf(ctx->sum_acc_mag, n) + stdf(ctx->sum_acc_mag, ctx->sum2_acc_mag, n);
131 for (uint16_t i = 1; i + 1 < ctx->count && i + 1 < (APP_IMU_EXPECTED_SAMPLES + 4); ++i) {
132 float a = ctx->acc_mag_hist[i - 1];
133 float b = ctx->acc_mag_hist[i];
134 float c = ctx->acc_mag_hist[i + 1];
135 if (b > thr && b > a && b >= c) peaks++;
136 }
137 out->peak_count = peaks;
138}
