47 const float ay = samples[i].acc_mg[1];
48 const float az = samples[i].acc_mg[2];
49 const float gx = samples[i].gyro_dps10[0];

56 acc_sq_sum[0] += ax * ax; acc_sq_sum[1] += ay * ay; acc_sq_sum[2] += az * az;

64 acc_mag_sq_sum += amag * amag;
65 gyro_mag_sum += gmag;

107 out->acc_sma = sat16f(acc_sma_sum * inv_n);

115 ++peaks;
116 }
117 }
