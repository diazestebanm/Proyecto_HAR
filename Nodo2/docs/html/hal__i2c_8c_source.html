144esp_err_t hal_i2c_read_reg(uint8_t dev_addr, uint8_t reg, uint8_t *data, size_t len)
145{
146 if (!data || len == 0) {
147 return ESP_ERR_INVALID_ARG;
148 }
149
150 i2c_cmd_handle_t cmd = i2c_cmd_link_create();
151 if (!cmd) {
152 return ESP_ERR_NO_MEM;
153 }
154
155 i2c_master_start(cmd);
156 i2c_master_write_byte(cmd, (dev_addr << 1) | I2C_MASTER_WRITE, true);
157 i2c_master_write_byte(cmd, reg, true);
158 i2c_master_start(cmd);
159 i2c_master_write_byte(cmd, (dev_addr << 1) | I2C_MASTER_READ, true);
160
161 if (len > 1) {
162 i2c_master_read(cmd, data, len - 1, I2C_MASTER_ACK);
163 }
164 i2c_master_read_byte(cmd, &data[len - 1], I2C_MASTER_LAST_NACK);
165
166 i2c_master_stop(cmd);
167
168 esp_err_t err = i2c_master_cmd_begin((i2c_port_t)APP_I2C_PORT, cmd, pdMS_TO_TICKS(100));
169 i2c_cmd_link_delete(cmd);
170 return err;
