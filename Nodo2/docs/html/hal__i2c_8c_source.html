50esp_err_t hal_i2c_write_reg(uint8_t dev_addr, uint8_t reg, uint8_t value)
51{
52 i2c_cmd_handle_t cmd = i2c_cmd_link_create();
53 if (!cmd) {
54 return ESP_ERR_NO_MEM;
55 }
56
57 i2c_master_start(cmd);
58 i2c_master_write_byte(cmd, (dev_addr << 1) | I2C_MASTER_WRITE, true);
59 i2c_master_write_byte(cmd, reg, true);
60 i2c_master_write_byte(cmd, value, true);
61 i2c_master_stop(cmd);
62 esp_err_t err = i2c_master_cmd_begin((i2c_port_t)APP_I2C_PORT, cmd, pdMS_TO_TICKS(100));
63 i2c_cmd_link_delete(cmd);
64 return err;
65}
