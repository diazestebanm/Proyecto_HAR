91static bool is_n1_matchable(uint32_t expected_rx_ms, radio_rx_event_t *out, uint32_t *delta_ms)
92{
93 bool ok = false;
94 xSemaphoreTake(s_n1_mutex, portMAX_DELAY);
95 if (s_last_n1_valid && s_last_n1.packet.seq != s_last_consumed_seq) {
96 const uint32_t rx_ms = s_last_n1.rx_local_ms;
97 const uint32_t diff = (rx_ms > expected_rx_ms) ? (rx_ms - expected_rx_ms) : (expected_rx_ms - rx_ms);
98 if (diff <= APP_MATCH_WINDOW_MS) {
99 if (out) *out = s_last_n1;
100 if (delta_ms) *delta_ms = diff;
101 s_last_consumed_seq = s_last_n1.packet.seq;
102 ok = true;
103 }
104 }
105 xSemaphoreGive(s_n1_mutex);
106 return ok;
107}

132static void task_imu(void *arg)
133{
134 (void)arg;
135 imu_sample_t samples[APP_IMU_WINDOW_SAMPLES];
136
137 while (1) {
138 uint32_t cmd = 0;
139 xTaskNotifyWait(0, 0xFFFFFFFFu, &cmd, portMAX_DELAY);
140
141 if (cmd == APP_IMU_CMD_CALIBRATE) {
142 esp_err_t err = hal_mpu6050_calibrate_gyro(&s_cal);
143 ESP_LOGI("IMU", "Calibracion gyro: %s", esp_err_to_name(err));
144 continue;
145 }
146
147 if (cmd == APP_IMU_CMD_CAPTURE) {
148 node2_window_result_t result = {
149 .boot_id = s_boot_id,
150 };
151 result.win_start_ms = app_now_ms() - s_t0_ms;
152 result.window_index = 0;
153
154 uint16_t captured = 0;
155 uint16_t flags = 0;
156 hal_mpu6050_capture_window(&s_cal, samples, APP_IMU_WINDOW_SAMPLES, &captured, &flags, s_t0_ms);
157
158 result.win_end_ms = app_now_ms() - s_t0_ms;
159 result.sample_count = captured;
160 result.flags = flags;
161 imu_features_compute(samples, captured, &result.imu);
162
163 xQueueOverwrite(s_imu_result_queue, &result);
164 }
165 }

171static void task_sd(void *arg)
172{
173 (void)arg;
174 log_record_t batch[APP_SD_BATCH_LINES];
175 size_t count = 0;
176 uint32_t last_flush_ms = app_now_ms();
177 log_record_t item;
178
179 while (1) {
180 if (xQueueReceive(s_sd_queue, &item, pdMS_TO_TICKS(APP_SD_FLUSH_PERIOD_MS)) == pdTRUE) {
181 if (count < APP_SD_BATCH_LINES) {
182 batch[count++] = item;
183 }
184 }
185
186 if (count == 0) {
187 continue;
188 }
189
190 if (count >= APP_SD_BATCH_LINES ||
191 (app_now_ms() - last_flush_ms) >= APP_SD_FLUSH_PERIOD_MS) {
192 if (hal_sdcard_is_ready()) {
193 hal_sdcard_append_records(batch, count);
194 }
195 count = 0;
196 last_flush_ms = app_now_ms();
197 }
198 }

205static esp_err_t initial_sync_wait(radio_rx_event_t *first)
206{
207 const uint32_t start_ms = app_now_ms();
208 while ((app_now_ms() - start_ms) < APP_NODE1_WAIT_TIMEOUT_MS) {
209 radio_rx_event_t ev;
210 uint32_t delta = 0;
211 if (is_n1_matchable(app_now_ms(), &ev, &delta)) {
212 (void)delta;
213 *first = ev;
214 return ESP_OK;
215 }
216 vTaskDelay(pdMS_TO_TICKS(100));
217 }
218 return ESP_ERR_TIMEOUT;
219}

227static void task_control_fsm(void *arg)
228{
229 (void)arg;
230 power_management_init();
231
232 battery_status_t bat;
233 hal_battery_sample(&bat);
234 s_bat = bat;
235 s_last_bat_ms = app_now_ms();
236
237 xTaskNotify(s_task_imu, APP_IMU_CMD_CALIBRATE, eSetValueWithOverwrite);
238 vTaskDelay(pdMS_TO_TICKS(3500));
239
240 radio_rx_event_t first = {0};
241 if (initial_sync_wait(&first) != ESP_OK) {
242 ESP_LOGW(TAG, "Nodo 1 aun no sincroniza. Esperando paquetes...");
243 while (initial_sync_wait(&first) != ESP_OK) {
244 vTaskDelay(pdMS_TO_TICKS(500));
245 }
246 }
247
248 uint32_t expected_next_rx_ms = first.rx_local_ms + APP_CYCLE_PERIOD_MS;
249 uint16_t window_index = 0;
250
251 while (1) {
252 const uint32_t now_ms = app_now_ms();
253 const uint32_t capture_start_ms = (expected_next_rx_ms > APP_CAPTURE_WINDOW_MS)
254 ? (expected_next_rx_ms - APP_CAPTURE_WINDOW_MS)
255 : now_ms;
256
257 if (capture_start_ms > now_ms) {
258 vTaskDelay(pdMS_TO_TICKS(capture_start_ms - now_ms));
259 }
260
261 if ((app_now_ms() - s_last_bat_ms) >= APP_BATTERY_SAMPLE_PERIOD_MS) {
262 hal_battery_sample(&s_bat);
263 s_last_bat_ms = app_now_ms();
264 }
265
266 xTaskNotify(s_task_imu, APP_IMU_CMD_CAPTURE, eSetValueWithOverwrite);
267
268 node2_window_result_t n2 = {0};
269 if (xQueueReceive(s_imu_result_queue, &n2, pdMS_TO_TICKS(APP_CAPTURE_WINDOW_MS + 2000)) != pdTRUE) {
270 ESP_LOGW(TAG, "No se recibio resultado local de IMU");
271 expected_next_rx_ms += APP_CYCLE_PERIOD_MS;
272 continue;
273 }
274 n2.window_index = ++window_index;
275 n2.flags |= s_bat.flags;
276
277 radio_rx_event_t n1 = {0};
278 uint32_t delta_ms = 0;
279 const uint32_t wait_deadline = expected_next_rx_ms + APP_MATCH_WINDOW_MS;
280 bool matched = false;
281 while (app_now_ms() <= wait_deadline) {
282 if (is_n1_matchable(expected_next_rx_ms, &n1, &delta_ms)) {
283 matched = true;
284 break;
285 }
286 vTaskDelay(pdMS_TO_TICKS(20));
287 }
288
289 if (!matched) {
290 ESP_LOGW(TAG, "Sin match N1 para win=%u. Se omite escritura.", n2.window_index);
291 expected_next_rx_ms += APP_CYCLE_PERIOD_MS;
292 continue;
293 }
294
295 n2.flags |= APP_FLAG_MATCH_OK;
296
297 log_record_t record = {
298 .session_id = s_session_id,
299 .match_delta_ms = delta_ms,
300 .rx_local_ms = n1.rx_local_ms,
301 .offset_est_ms = (int32_t)n1.rx_local_ms - (int32_t)n1.packet.t_rel_ms,
302 .bat = s_bat,
303 .n2 = n2,
304 .n1 = n1,
305 };
306
307 xQueueSend(s_sd_queue, &record, portMAX_DELAY);
308 ESP_LOGI(TAG, "MATCH OK win=%u seq_n1=%u delta=%lu ms bat=%u%%",
309 n2.window_index, n1.packet.seq, (unsigned long)delta_ms, s_bat.percent);
310
311 /*
312 * Se reajusta el siguiente instante esperado usando el tiempo real del
313 * paquete recibido. Esto corrige deriva entre nodos sin requerir ACK.
314 */
315 expected_next_rx_ms = n1.rx_local_ms + APP_CYCLE_PERIOD_MS;
316 }
317}

323esp_err_t app_tasks_start(void)
324{
325 s_boot_id = (uint16_t)(esp_random() & 0xFFFF);
326 s_session_id = (uint32_t)esp_random();
327 s_t0_ms = app_now_ms();
328
329 s_radio_rx_queue = xQueueCreate(8, sizeof(radio_rx_event_t));
330 s_imu_result_queue = xQueueCreate(1, sizeof(node2_window_result_t));
331 s_sd_queue = xQueueCreate(8, sizeof(log_record_t));
332 s_n1_mutex = xSemaphoreCreateMutex();
333
334 if (!s_radio_rx_queue || !s_imu_result_queue || !s_sd_queue || !s_n1_mutex) {
335 return ESP_ERR_NO_MEM;
336 }
337
338 xTaskCreatePinnedToCore(task_imu, "TaskIMU", APP_TASK_STACK_DEFAULT, NULL, APP_TASK_PRIO_IMU, &s_task_imu, 1);
339
340 ESP_RETURN_ON_ERROR(hal_mpu6050_init(s_task_imu), TAG, "mpu init fallo");
341 ESP_RETURN_ON_ERROR(hal_mpu6050_verify(), TAG, "mpu verify fallo");
342 ESP_RETURN_ON_ERROR(hal_battery_init(), TAG, "battery init fallo");
343 (void)hal_sdcard_init(s_boot_id);
344 ESP_RETURN_ON_ERROR(hal_radio_init(s_radio_rx_queue), TAG, "radio init fallo");
345
346 xTaskCreatePinnedToCore(task_radio, "TaskRadioRx", APP_TASK_STACK_DEFAULT, NULL, APP_TASK_PRIO_RADIO, &s_task_radio, 0);
347 xTaskCreatePinnedToCore(task_sd, "TaskSD", APP_TASK_STACK_SD, NULL, APP_TASK_PRIO_SD, &s_task_sd, 1);
348 xTaskCreatePinnedToCore(task_control_fsm, "TaskControlFSM", APP_TASK_STACK_DEFAULT, NULL, APP_TASK_PRIO_CONTROL, &s_task_ctrl, 1);
349
350 ESP_LOGI(TAG, "Nodo 2 listo. boot_id=%u session_id=%lu", s_boot_id, (unsigned long)s_session_id);
351 return ESP_OK;
352}
