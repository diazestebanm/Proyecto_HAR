80static bool is_n1_matchable(uint32_t expected_rx_ms, radio_rx_event_t *out, uint32_t *delta_ms)
81{
82 bool ok = false;
83 xSemaphoreTake(s_n1_mutex, portMAX_DELAY);
84 if (s_last_n1_valid && s_last_n1.packet.seq != s_last_consumed_seq) {
85 const uint32_t rx_ms = s_last_n1.rx_local_ms;
86 const uint32_t diff = (rx_ms > expected_rx_ms) ? (rx_ms - expected_rx_ms) : (expected_rx_ms - rx_ms);
87 if (diff <= APP_MATCH_WINDOW_MS) {
88 if (out) *out = s_last_n1;
89 if (delta_ms) *delta_ms = diff;
90 s_last_consumed_seq = s_last_n1.packet.seq;
91 ok = true;
92 }
93 }
94 xSemaphoreGive(s_n1_mutex);
95 return ok;
96}

101static void task_radio(void *arg)
102{
103 (void)arg;
104 radio_rx_event_t ev;
105
106 while (1) {
107 if (xQueueReceive(s_radio_rx_queue, &ev, portMAX_DELAY) == pdTRUE) {
108 if (ev.packet.boot_id == s_last_seen_boot &&
109 ev.packet.seq == s_last_seen_seq) {
110 continue;
111 }
112
113 s_last_seen_boot = ev.packet.boot_id;
114 s_last_seen_seq = ev.packet.seq;
115
116 xSemaphoreTake(s_n1_mutex, portMAX_DELAY);
117 s_last_n1 = ev;
118 s_last_n1_valid = true;
119 xSemaphoreGive(s_n1_mutex);
120
121 ESP_LOGI("RADIO", "N1 seq=%u type=%u rx_local_ms=%lu",
122 ev.packet.seq, ev.packet.msg_type, (unsigned long)ev.rx_local_ms);
123 }

130static void task_imu(void *arg)
131{
132 (void)arg;
133 imu_sample_t samples[APP_IMU_WINDOW_SAMPLES];
134
135 while (1) {
136 uint32_t cmd = 0;
137 xTaskNotifyWait(0, 0xFFFFFFFFu, &cmd, portMAX_DELAY);
138
139 if (cmd == APP_IMU_CMD_CALIBRATE) {
140 esp_err_t err = hal_mpu6050_calibrate_gyro(&s_cal);
141 if (err != ESP_OK) {
142 memset(&s_cal, 0, sizeof(s_cal));
143 }
144 ESP_LOGI("IMU", "Calibracion gyro: %s", esp_err_to_name(err));
145 continue;
146 }
147
148 if (cmd == APP_IMU_CMD_CAPTURE) {
149 node2_window_result_t result = {
150 .boot_id = s_boot_id,
151 };
152 result.win_start_ms = app_now_ms() - s_t0_ms;
153 result.window_index = 0;
154
155 uint16_t captured = 0;
156 uint16_t flags = 0;
157 esp_err_t err = hal_mpu6050_capture_window(&s_cal, samples, APP_IMU_WINDOW_SAMPLES, &captured, &flags, s_t0_ms);
158 if (err != ESP_OK) {
159 ESP_LOGE("IMU", "Captura IMU fallo: %s", esp_err_to_name(err));
160
161 if (hal_mpu6050_recover() == ESP_OK) {
162 memset(&s_cal, 0, sizeof(s_cal));
163 if (hal_mpu6050_calibrate_gyro(&s_cal) == ESP_OK) {
164 ESP_LOGW("IMU", "MPU recuperado y recalibrado");
165 }
166 }
167
168 continue;
169 }
170
171 result.win_end_ms = app_now_ms() - s_t0_ms;
172 result.sample_count = captured;
173 result.flags = flags;
174 imu_features_compute(samples, captured, &result.imu);
175
176 xQueueOverwrite(s_imu_result_queue, &result);
177 }
178 }

184static void task_sd(void *arg)
185{
186 (void)arg;
187 log_record_t batch[APP_SD_BATCH_LINES];
188 size_t count = 0;
189 uint32_t last_flush_ms = app_now_ms();
190 log_record_t item;
191
192 while (1) {
193 if (xQueueReceive(s_sd_queue, &item, pdMS_TO_TICKS(APP_SD_FLUSH_PERIOD_MS)) == pdTRUE) {
194 if (count < APP_SD_BATCH_LINES) {
195 batch[count++] = item;
196 }
197 }
198
199 if (count == 0) {
200 continue;
201 }
202
203 if (count >= APP_SD_BATCH_LINES ||
204 (app_now_ms() - last_flush_ms) >= APP_SD_FLUSH_PERIOD_MS) {
205 if (hal_sdcard_is_ready()) {
206 esp_err_t err = hal_sdcard_append_records(batch, count);
207 if (err != ESP_OK) {
208 ESP_LOGE(TAG, "Persistencia host fallo: %s", esp_err_to_name(err));
209 }
210 }
211 count = 0;
212 last_flush_ms = app_now_ms();
213 }
214 }

221static esp_err_t initial_sync_wait(radio_rx_event_t *first)
222{
223 const uint32_t start_ms = app_now_ms();
224 while ((app_now_ms() - start_ms) < APP_NODE1_WAIT_TIMEOUT_MS) {
225 radio_rx_event_t ev;
226 uint32_t delta = 0;
227 if (is_n1_matchable(app_now_ms(), &ev, &delta)) {
228 (void)delta;
229 *first = ev;
230 return ESP_OK;
231 }
232 vTaskDelay(pdMS_TO_TICKS(100));
233 }
234 return ESP_ERR_TIMEOUT;
235}

323static void task_control_fsm(void *arg)
324{
325 (void)arg;
326
327
328 battery_status_t bat;
329 if (hal_battery_sample(&bat) == ESP_OK && bat.valid) {
330 s_bat = bat;
331 }
332 s_last_bat_ms = app_now_ms();
333
334 xTaskNotify(s_task_imu, APP_IMU_CMD_CALIBRATE, eSetValueWithOverwrite);
335 vTaskDelay(pdMS_TO_TICKS(3500));
336
337 radio_rx_event_t first = {0};
338 if (initial_sync_wait(&first) != ESP_OK) {
339 ESP_LOGW(TAG, "Nodo 1 aun no sincroniza. Esperando paquetes...");
340 while (initial_sync_wait(&first) != ESP_OK) {
341 vTaskDelay(pdMS_TO_TICKS(500));
342 }
343 }
344
345 uint32_t expected_next_rx_ms = first.rx_local_ms + APP_CYCLE_PERIOD_MS;
346 uint16_t window_index = 0;
347
348 while (1) {
349 const uint32_t now_ms = app_now_ms();
350 const uint32_t capture_start_ms = (expected_next_rx_ms > APP_CAPTURE_WINDOW_MS)
351 ? (expected_next_rx_ms - APP_CAPTURE_WINDOW_MS)
352 : now_ms;
353
354 if (capture_start_ms > now_ms) {
355 vTaskDelay(pdMS_TO_TICKS(capture_start_ms - now_ms));
356 }
357
358 if ((app_now_ms() - s_last_bat_ms) >= APP_BATTERY_SAMPLE_PERIOD_MS) {
359 battery_status_t bat_now;
360 if (hal_battery_sample(&bat_now) == ESP_OK && bat_now.valid) {
361 s_bat = bat_now;
362 }
363 s_last_bat_ms = app_now_ms();
364 }
365
366 xTaskNotify(s_task_imu, APP_IMU_CMD_CAPTURE, eSetValueWithOverwrite);
367
368 node2_window_result_t n2 = {0};
369 if (xQueueReceive(s_imu_result_queue, &n2, pdMS_TO_TICKS(APP_CAPTURE_WINDOW_MS + 2000)) != pdTRUE) {
370 ESP_LOGW(TAG, "No se recibio resultado local de IMU");
371 expected_next_rx_ms += APP_CYCLE_PERIOD_MS;
372 continue;
373 }
374 n2.window_index = ++window_index;
375 n2.flags |= s_bat.flags;
376
377 radio_rx_event_t n1 = {0};
378 uint32_t delta_ms = 0;
379 const uint32_t wait_deadline = expected_next_rx_ms + APP_MATCH_WINDOW_MS;
380 bool matched = false;
381 while (app_now_ms() <= wait_deadline) {
382 if (is_n1_matchable(expected_next_rx_ms, &n1, &delta_ms)) {
383 matched = true;
384 break;
385 }
386 vTaskDelay(pdMS_TO_TICKS(20));
387 }
388
389 if (!matched) {
390 ESP_LOGW(TAG, "Sin match N1 para win=%u. Se omite escritura.", n2.window_index);
391 expected_next_rx_ms += APP_CYCLE_PERIOD_MS;
392 continue;
393 }
394
395 n2.flags |= APP_FLAG_MATCH_OK;
396
397 log_record_t record = {
398 .session_id = s_session_id,
399 .match_delta_ms = delta_ms,
400 .rx_local_ms = n1.rx_local_ms,
401 .offset_est_ms = (int32_t)n1.rx_local_ms - (int32_t)n1.packet.t_rel_ms,
402 .bat = s_bat,
403 .n2 = n2,
404 .n1 = n1,
405 };
406
407 #if APP_SERIAL_SUMMARY_ENABLE
408 if ((record.n2.window_index % APP_SERIAL_SUMMARY_EVERY_N_WINDOWS) == 0U) {
409 log_fused_window_summary(&record);
410 }
411 #endif
412
413 xQueueSend(s_sd_queue, &record, portMAX_DELAY);
414
415 /*
416 * Se reajusta el siguiente instante esperado usando el tiempo real del
417 * paquete recibido. Esto corrige deriva entre nodos sin requerir ACK.
418 */
419 expected_next_rx_ms = n1.rx_local_ms + APP_CYCLE_PERIOD_MS;
420 }
421}

427esp_err_t app_tasks_start(void)
428{
429 s_boot_id = (uint16_t)(esp_random() & 0xFFFF);
430 s_session_id = (uint32_t)esp_random();
431 s_t0_ms = app_now_ms();
432
433 s_radio_rx_queue = xQueueCreate(8, sizeof(radio_rx_event_t));
434 s_imu_result_queue = xQueueCreate(1, sizeof(node2_window_result_t));
435 s_sd_queue = xQueueCreate(8, sizeof(log_record_t));
436 s_n1_mutex = xSemaphoreCreateMutex();
437
438 if (!s_radio_rx_queue || !s_imu_result_queue || !s_sd_queue || !s_n1_mutex) {
439 return ESP_ERR_NO_MEM;
440 }
441
442 xTaskCreatePinnedToCore(task_imu, "TaskIMU", APP_TASK_STACK_DEFAULT, NULL, APP_TASK_PRIO_IMU, &s_task_imu, 1);
443
444 esp_err_t mpu_err = ESP_FAIL;
445 for (int i = 0; i < 5; ++i) {
446 mpu_err = hal_mpu6050_init(s_task_imu);
447 if (mpu_err == ESP_OK) {
448 mpu_err = hal_mpu6050_verify();
449 if (mpu_err == ESP_OK) {
450 break;
451 }
452 }
453
454 ESP_LOGW(TAG, "Reintentando init MPU (%d/5): %s", i + 1, esp_err_to_name(mpu_err));
455 vTaskDelay(pdMS_TO_TICKS(300));
456 }
457 ESP_RETURN_ON_ERROR(mpu_err, TAG, "mpu init fallo");
458 ESP_RETURN_ON_ERROR(hal_battery_init(), TAG, "battery init fallo");
459 ESP_RETURN_ON_ERROR(hal_sdcard_init(s_boot_id), TAG, "host logger init fallo");
460 ESP_RETURN_ON_ERROR(hal_radio_init(s_radio_rx_queue), TAG, "radio init fallo");
461
462 xTaskCreatePinnedToCore(task_radio, "TaskRadioRx", APP_TASK_STACK_DEFAULT, NULL, APP_TASK_PRIO_RADIO, &s_task_radio, 0);
463 xTaskCreatePinnedToCore(task_sd, "TaskSD", APP_TASK_STACK_SD, NULL, APP_TASK_PRIO_SD, &s_task_sd, 1);
464 xTaskCreatePinnedToCore(task_control_fsm, "TaskControlFSM", APP_TASK_STACK_DEFAULT, NULL, APP_TASK_PRIO_CONTROL, &s_task_ctrl, 1);
465
466 ESP_LOGI(TAG, "Nodo 2 listo. boot_id=%u session_id=%lu", s_boot_id, (unsigned long)s_session_id);
467 return ESP_OK;
468}
