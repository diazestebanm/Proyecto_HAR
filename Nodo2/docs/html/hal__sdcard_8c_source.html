87esp_err_t hal_sdcard_init(uint16_t boot_id)
88{
89 esp_log_level_set("sdspi_transaction", ESP_LOG_WARN);
90 vTaskDelay(pdMS_TO_TICKS(200));
91
92 s_host = (sdmmc_host_t)SDSPI_HOST_DEFAULT();
93 s_host.max_freq_khz = SDMMC_FREQ_PROBING;
94
95 spi_bus_config_t bus_cfg = {
96 .mosi_io_num = APP_SD_MOSI_GPIO,
97 .miso_io_num = APP_SD_MISO_GPIO,
98 .sclk_io_num = APP_SD_SCK_GPIO,
99 .quadwp_io_num = -1,
100 .quadhd_io_num = -1,
101 .max_transfer_sz = 4096,
102 };
103 ESP_RETURN_ON_ERROR(spi_bus_initialize(s_host.slot, &bus_cfg, SDSPI_DEFAULT_DMA), TAG, "spi bus init fallo");
104
105 sdspi_device_config_t dev_cfg = SDSPI_DEVICE_CONFIG_DEFAULT();
106 dev_cfg.gpio_cs = APP_SD_CS_GPIO;
107 dev_cfg.host_id = s_host.slot;
108
109 esp_vfs_fat_sdmmc_mount_config_t mount_cfg = {
110 .format_if_mount_failed = false,
111 .max_files = 4,
112 .allocation_unit_size = 16 * 1024,
113 };
114
115 esp_err_t err = esp_vfs_fat_sdspi_mount(APP_SD_MOUNT_POINT, &s_host, &dev_cfg, &mount_cfg, &s_card);
116 if (err != ESP_OK) {
117 ESP_LOGW(TAG, "No se pudo montar SD: %s", esp_err_to_name(err));
118 s_ready = false;
119 return err;
120 }
121
122 snprintf(s_path, sizeof(s_path), APP_SD_MOUNT_POINT "/adl_session_%04X.csv", boot_id);
123 FILE *f = fopen(s_path, "r");
124 if (f) {
125 fclose(f);
126 } else {
127 f = fopen(s_path, "w");
128 if (!f) {
129 s_ready = false;
130 return ESP_FAIL;
131 }
132 csv_write_header(f);
133 fclose(f);
134 }
135
136 s_ready = true;
137 ESP_LOGI(TAG, "SD lista. Archivo: %s", s_path);
138 return ESP_OK;
139}

152esp_err_t hal_sdcard_append_records(const log_record_t *records, size_t count)
153{
154 if (!s_ready || !records || count == 0) {
155 return ESP_ERR_INVALID_STATE;
156 }
157
158 FILE *f = fopen(s_path, "a");
159 if (!f) {
160 s_ready = false;
161 return ESP_FAIL;
162 }
163
164 for (size_t i = 0; i < count; ++i) {
165 const log_record_t *r = &records[i];
166
167 fprintf(f,
168 "%lu,%lu,%lu,%ld,%u,%u,%lu,%lu,%u,%u,%u,%.3f,",
169 (unsigned long)r->session_id,
170 (unsigned long)r->match_delta_ms,
171 (unsigned long)r->rx_local_ms,
172 (long)r->offset_est_ms,
173 r->n2.boot_id,
174 r->n2.window_index,
175 (unsigned long)r->n2.win_start_ms,
176 (unsigned long)r->n2.win_end_ms,
177 r->n2.sample_count,
178 r->n2.flags,
179 r->bat.percent,
180 r->bat.voltage_v);
181
182 csv_write_imu(f, &r->n2.imu);
183
184 fprintf(f,
185 ",%u,%u,%lu,%u,%u,%u,%u,%u,%u,",
186 r->n1.packet.boot_id,
187 r->n1.packet.seq,
188 (unsigned long)r->n1.packet.t_rel_ms,
189 r->n1.packet.window_ms,
190 r->n1.packet.sample_count,
191 r->n1.packet.flags,
192 r->n1.packet.bpm,
193 r->n1.packet.spo2,
194 r->n1.packet.battery_pct);
195
196 /*
197 * node1_packet_t esta marcado como packed para mantener compatibilidad
198 * binaria con el payload recibido por ESP-NOW. Para evitar el warning
199 * por direccion de miembro packed, se copia el resumen IMU a una
200 * variable local alineada antes de pasarlo al escritor CSV.
201 */
202 imu_feature_summary_t n1_imu;
203 n1_packet_to_imu_summary(&r->n1.packet, &n1_imu);
204 csv_write_imu(f, &n1_imu);
205 fputc('\n', f);
206 }
207
208 fclose(f);
209 return ESP_OK;
210}
