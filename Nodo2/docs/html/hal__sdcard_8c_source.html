39static esp_err_t buf_append(char *dst, size_t dst_size, size_t *used, const char *fmt, ...)
40{
41 if (!dst || !used || !fmt || *used >= dst_size) {
42 return ESP_ERR_INVALID_ARG;
43 }
44
45 va_list args;
46 va_start(args, fmt);
47 int written = vsnprintf(dst + *used, dst_size - *used, fmt, args);
48 va_end(args);
49
50 if (written < 0) {
51 return ESP_FAIL;
52 }
53
54 size_t w = (size_t)written;
55 if (w >= (dst_size - *used)) {
56 return ESP_ERR_INVALID_SIZE;
57 }
58
59 *used += w;
60 return ESP_OK;
61}

147static esp_err_t csv_build_record_line(const log_record_t *r, char *line, size_t line_size)
148{
149 ESP_RETURN_ON_FALSE(r != NULL, ESP_ERR_INVALID_ARG, TAG, "record nulo");

162 r->n2.boot_id,

165 r->n2.win_end_ms,

167 r->n2.flags,
168 r->bat.percent,
169 (double)r->bat.voltage_v),
170 TAG, "encabezado de record no cabe");
171
172 ESP_RETURN_ON_ERROR(csv_append_imu(line, line_size, &used, &r->n2.imu), TAG, "imu n2 invalida");
173
174 ESP_RETURN_ON_ERROR(buf_append(line, line_size, &used,
175 ",%u,%u,%" PRIu32 ",%u,%u,%u,%u,%u,%u,",

177 r->n1.packet.seq,
178 r->n1.packet.t_rel_ms,
179 r->n1.packet.window_ms,
180 r->n1.packet.sample_count,
181 r->n1.packet.flags,
182 r->n1.packet.bpm,
183 r->n1.packet.spo2,
184 r->n1.packet.battery_pct),
185 TAG, "encabezado n1 no cabe");
186
187 imu_feature_summary_t n1_imu;
188 n1_packet_to_imu_summary(&r->n1.packet, &n1_imu);
189 ESP_RETURN_ON_ERROR(csv_append_imu(line, line_size, &used, &n1_imu), TAG, "imu n1 invalida");

197esp_err_t hal_sdcard_init(uint16_t boot_id)
198{
199 char header[APP_LOG_LINE_MAX] = {0};
200
201 snprintf(s_filename, sizeof(s_filename), "adl_session_%04X.csv", boot_id);
202 ESP_RETURN_ON_ERROR(csv_build_header(header, sizeof(header)), TAG, "header csv demasiado grande");
203
204 s_ready = true;
205
206 host_emit_line(HOST_META_PREFIX, s_filename);
207 host_emit_line(HOST_HEADER_PREFIX, header);
208 ESP_LOGI(TAG, "Backend serial listo. Archivo destino en host: %s", s_filename);
209 return ESP_OK;

223esp_err_t hal_sdcard_append_records(const log_record_t *records, size_t count)
224{
225 if (!s_ready || !records || count == 0) {
226 return ESP_ERR_INVALID_STATE;
227 }
228
