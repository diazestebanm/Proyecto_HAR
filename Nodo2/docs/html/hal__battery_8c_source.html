73static float vbat_to_percent(float v)
74{
75 static const float map_v[] = {4.20f,4.10f,4.00f,3.90f,3.80f,3.70f,3.60f,3.50f,3.40f,3.30f,3.20f,3.10f,3.00f};
76 static const float map_pct[] = {100.f,95.f,88.f,78.f,65.f,50.f,36.f,24.f,14.f,8.f,4.f,1.f,0.f};
77 const int n = 13;
78
79 if (v >= map_v[0]) return 100.0f;
80 if (v <= map_v[n - 1]) return 0.0f;
81
82 for (int i = 0; i < n - 1; ++i) {
83 if (v >= map_v[i + 1]) {
84 const float t = (v - map_v[i + 1]) / (map_v[i] - map_v[i + 1]);
85 return map_pct[i + 1] + t * (map_pct[i] - map_pct[i + 1]);
86 }
87 }
88 return 0.0f;
89}

117esp_err_t hal_battery_sample(battery_status_t *out)
118{
119 if (!out) {
120 return ESP_ERR_INVALID_ARG;
121 }
122 memset(out, 0, sizeof(*out));
123
124 int samples[APP_BAT_NUM_SAMPLES];
125 for (int i = 0; i < APP_BAT_NUM_SAMPLES; ++i) {
126 ESP_RETURN_ON_ERROR(adc_oneshot_read(s_adc, APP_BAT_ADC_CHANNEL, &samples[i]), TAG, "adc read fallo");
127 vTaskDelay(pdMS_TO_TICKS(2));
128 }
129
130 qsort(samples, APP_BAT_NUM_SAMPLES, sizeof(int), cmp_int);
131 const int range = samples[APP_BAT_NUM_SAMPLES - 1] - samples[0];
132 const int median_raw = (samples[APP_BAT_NUM_SAMPLES / 2] + samples[(APP_BAT_NUM_SAMPLES / 2) - 1]) / 2;
133
134 if (range > APP_BAT_FLOAT_RANGE_THRESH) {
135 s_ema_vbat = -1.0f;
136 s_last_pct = -1.0f;
137 return ESP_OK;
138 }
139
140 const float v_gpio = raw_to_mv(median_raw) / 1000.0f;
141 const float v_bat = v_gpio * APP_BAT_DIVIDER_RATIO;
142
143 if (s_ema_vbat < 0.0f) {
144 s_ema_vbat = v_bat;
145 } else {
146 s_ema_vbat = APP_BAT_EMA_ALPHA * v_bat + (1.0f - APP_BAT_EMA_ALPHA) * s_ema_vbat;
147 }
148
149 if (s_ema_vbat < APP_BAT_MIN_VALID_V || s_ema_vbat > APP_BAT_MAX_VALID_V) {
150 return ESP_OK;
151 }
152
153 const float new_pct = vbat_to_percent(s_ema_vbat);
154 if (s_last_pct < 0.0f || fabsf(new_pct - s_last_pct) >= APP_BAT_PCT_HYSTERESIS) {
155 s_last_pct = new_pct;
156 }
157
158 out->valid = true;
159 out->voltage_v = s_ema_vbat;
160 out->percent = (uint8_t)lroundf(fmaxf(0.0f, fminf(100.0f, s_last_pct)));
161 out->low = (s_ema_vbat < APP_BAT_LOW_V);
162 out->critical = (s_ema_vbat < APP_BAT_CRITICAL_V);
163 out->flags |= APP_FLAG_BAT_VALID;
164 if (out->low) out->flags |= APP_FLAG_BAT_LOW;
165 if (out->critical) out->flags |= APP_FLAG_BAT_CRITICAL;
166
167 return ESP_OK;
168}
