45static void radio_rx_cb(const esp_now_recv_info_t *info, const uint8_t *data, int len)
46{
47 if (!info || !data || len != (int)sizeof(node1_packet_t)) {
48 return;
49 }
50 if (!node1_mac_allowed(info->src_addr)) {
51 return;
52 }
53
54 radio_rx_event_t ev = {0};
55 memcpy(ev.src_mac, info->src_addr, 6);
56 memcpy(&ev.packet, data, sizeof(node1_packet_t));
57 ev.rx_local_ms = (uint32_t)(esp_timer_get_time() / 1000ULL);
58
59 if (s_rx_queue) {
60 xQueueSend(s_rx_queue, &ev, 0);
61 }
62}

68esp_err_t hal_radio_init(QueueHandle_t rx_queue)
69{
70 s_rx_queue = rx_queue;
71
72 esp_err_t err = nvs_flash_init();
73 if (err == ESP_ERR_NVS_NO_FREE_PAGES || err == ESP_ERR_NVS_NEW_VERSION_FOUND) {
74 ESP_ERROR_CHECK(nvs_flash_erase());
75 err = nvs_flash_init();
76 }
77 ESP_RETURN_ON_ERROR(err, TAG, "nvs init fallo");
78
79 ESP_RETURN_ON_ERROR(esp_netif_init(), TAG, "esp_netif_init fallo");
80 ESP_RETURN_ON_ERROR(esp_event_loop_create_default(), TAG, "event loop fallo");
81 esp_netif_create_default_wifi_sta();
82
83 wifi_init_config_t wifi_cfg = WIFI_INIT_CONFIG_DEFAULT();
84 ESP_RETURN_ON_ERROR(esp_wifi_init(&wifi_cfg), TAG, "wifi init fallo");
85 ESP_RETURN_ON_ERROR(esp_wifi_set_mode(WIFI_MODE_STA), TAG, "wifi mode fallo");
86 ESP_RETURN_ON_ERROR(esp_wifi_set_storage(WIFI_STORAGE_RAM), TAG, "wifi storage fallo");
87 ESP_RETURN_ON_ERROR(esp_wifi_start(), TAG, "wifi start fallo");
88 ESP_RETURN_ON_ERROR(esp_wifi_set_channel(APP_WIFI_CHANNEL, WIFI_SECOND_CHAN_NONE), TAG, "wifi channel fallo");
89
90 /*
91 * Nodo 2 actua como receptor, asi que no registra peers de salida.
92 * Solo necesita que el stack Wi-Fi este levantado y ESP-NOW inicializado.
93 */
94 ESP_RETURN_ON_ERROR(esp_now_init(), TAG, "esp_now_init fallo");
95 ESP_RETURN_ON_ERROR(esp_now_register_recv_cb(radio_rx_cb), TAG, "register recv cb fallo");
96
97#if APP_USE_ESPNOW_RX_PS
98 esp_now_set_wake_window(APP_ESPNOW_WAKE_WINDOW_MS);
99 esp_wifi_connectionless_module_set_wake_interval(APP_ESPNOW_WAKE_INTERVAL_MS);
100#endif
101
102 uint8_t mac[6] = {0};
103 esp_wifi_get_mac(WIFI_IF_STA, mac);
104 ESP_LOGI(TAG, "ESP-NOW RX listo. CH=%d MAC=%02X:%02X:%02X:%02X:%02X:%02X",
105 APP_WIFI_CHANNEL,
106 mac[0], mac[1], mac[2], mac[3], mac[4], mac[5]);
107 return ESP_OK;
108}
