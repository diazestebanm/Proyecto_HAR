97static esp_err_t mpu_set_awake(bool awake)
98{
99 esp_err_t err = ESP_FAIL;
100
101 for (int attempt = 0; attempt < 3; ++attempt) {
102 if (awake) {
103 err = mpu_write(REG_PWR_MGMT_1, MPU_CLKSEL_PLL_XGYRO);
104 if (err == ESP_OK) {
105 err = mpu_write(REG_PWR_MGMT_2, 0x00);
106 }
107 } else {
108 err = mpu_write(REG_INT_ENABLE, 0x00);
109 if (err == ESP_OK) {
110 err = mpu_write(REG_PWR_MGMT_1, MPU_SLEEP_BIT);
111 }
112 }
113
114 if (err == ESP_OK) {
115 vTaskDelay(pdMS_TO_TICKS(20));
116 return ESP_OK;
117 }
118
119 vTaskDelay(pdMS_TO_TICKS(30));
120 }
121
122 return err;
123}

131static esp_err_t mpu_probe_one(uint8_t addr, uint8_t *who_out)
132{
133 uint8_t who = 0;
134 s_mpu_addr = addr;
135
136 if (hal_i2c_ping(addr) != ESP_OK) {
137 return ESP_ERR_NOT_FOUND;
138 }
139
140 if (mpu_read(REG_WHO_AM_I, &who, 1) != ESP_OK) {
141 return ESP_FAIL;
142 }
143
144 if (who == 0x00 || who == 0xFF) {
145 return ESP_ERR_NOT_FOUND;
146 }
147
148 if (who_out) {
149 *who_out = who;
150 }
151 return ESP_OK;
152}

160static esp_err_t mpu_probe_and_select(uint8_t *who_out)
161{
162 esp_err_t err;
163 uint8_t who = 0;
164
165 err = mpu_probe_one(APP_MPU_ADDR, &who);
166 if (err == ESP_OK) {
167 if (who_out) *who_out = who;
168 return ESP_OK;
169 }
170
171 /* Probar la direccion alternativa por si AD0 cambia o el modulo la usa */
172 const uint8_t alt = (APP_MPU_ADDR == 0x68) ? 0x69 : 0x68;
173 err = mpu_probe_one(alt, &who);
174 if (err == ESP_OK) {
175 ESP_LOGW(TAG, "MPU detectado en direccion alternativa 0x%02X", alt);
176 if (who_out) *who_out = who;
177 return ESP_OK;
178 }
179
180 return ESP_ERR_NOT_FOUND;
181}

189static esp_err_t mpu_soft_reset_and_configure(void)
190{
191 ESP_RETURN_ON_ERROR(mpu_write(REG_PWR_MGMT_1, MPU_SOFT_RESET), TAG, "soft reset fallo");
192 vTaskDelay(pdMS_TO_TICKS(120));
193
194 ESP_RETURN_ON_ERROR(mpu_set_awake(true), TAG, "despertar fallo");
195 vTaskDelay(pdMS_TO_TICKS(50));
196
197 /*
198 * DLPF moderado y 25 Hz de salida. Con DLPF activado la base es 1 kHz,
199 * por eso: rate = 1000 / (1 + SMPLRT_DIV) => 25 Hz -> div = 39.
200 */
201 ESP_RETURN_ON_ERROR(mpu_write(REG_CONFIG, 0x03), TAG, "config dlpf fallo");
202 ESP_RETURN_ON_ERROR(mpu_write(REG_SMPLRT_DIV, 39), TAG, "smplrt_div fallo");
203 ESP_RETURN_ON_ERROR(mpu_write(REG_GYRO_CONFIG, 0x00), TAG, "gyro config fallo");
204 ESP_RETURN_ON_ERROR(mpu_write(REG_ACCEL_CONFIG, 0x00), TAG, "accel config fallo");
205 ESP_RETURN_ON_ERROR(mpu_write(REG_FIFO_EN, 0x00), TAG, "fifo en fallo");
206 ESP_RETURN_ON_ERROR(mpu_write(REG_INT_PIN_CFG, 0x00), TAG, "int pin cfg fallo");
207 ESP_RETURN_ON_ERROR(mpu_write(REG_INT_ENABLE, 0x00), TAG, "int disable fallo");
208
209 return ESP_OK;
210}

219esp_err_t hal_mpu6050_init(TaskHandle_t imu_task_to_notify)
220{
221 s_imu_task = imu_task_to_notify;
222 ESP_RETURN_ON_FALSE(s_imu_task != NULL, ESP_ERR_INVALID_ARG, TAG, "imu task nula");
223
224 if (!s_drdy_sem) {
225 s_drdy_sem = xSemaphoreCreateBinary();
226 ESP_RETURN_ON_FALSE(s_drdy_sem != NULL, ESP_ERR_NO_MEM, TAG, "sem drdy no creada");
227 }
228
229 while (xSemaphoreTake(s_drdy_sem, 0) == pdTRUE) {
230 }
231
232 ESP_RETURN_ON_ERROR(hal_i2c_init(), TAG, "I2C no disponible");
233
234 /* Dar tiempo real al modulo para estabilizarse despues del power-up */
235 vTaskDelay(pdMS_TO_TICKS(200));
236
237 gpio_config_t io = {
238 .pin_bit_mask = (1ULL << APP_MPU_INT_GPIO),
239 .mode = GPIO_MODE_INPUT,
240 .pull_up_en = GPIO_PULLUP_DISABLE,
241 .pull_down_en = GPIO_PULLDOWN_DISABLE,
242 .intr_type = GPIO_INTR_POSEDGE,
243 };
244 ESP_RETURN_ON_ERROR(gpio_config(&io), TAG, "gpio int config fallo");
245
246 if (!s_isr_service_installed) {
247 esp_err_t isr_err = gpio_install_isr_service(0);
248 if (isr_err == ESP_OK || isr_err == ESP_ERR_INVALID_STATE) {
249 s_isr_service_installed = true;
250 } else {
251 return isr_err;
252 }
253 }
254
255 gpio_isr_handler_remove(APP_MPU_INT_GPIO);
256 ESP_RETURN_ON_ERROR(gpio_isr_handler_add(APP_MPU_INT_GPIO, mpu_isr_handler, NULL), TAG, "agregar isr fallo");
257
258 esp_err_t err = ESP_FAIL;
259 uint8_t who = 0;
260
261 for (int attempt = 0; attempt < 5; ++attempt) {
262 ESP_LOGW(TAG, "Intento init MPU %d/5", attempt + 1);
263
264 hal_i2c_deinit();
265 ESP_RETURN_ON_ERROR(hal_i2c_init(), TAG, "reinit i2c fallo");
266 vTaskDelay(pdMS_TO_TICKS(100));
267
268 err = mpu_probe_and_select(&who);
269 if (err != ESP_OK) {
270 vTaskDelay(pdMS_TO_TICKS(120));
271 continue;
272 }
273
274 err = mpu_soft_reset_and_configure();
275 if (err == ESP_OK) {
276 while (xSemaphoreTake(s_drdy_sem, 0) == pdTRUE) {
277 }
278 ESP_LOGI(TAG, "MPU6050 listo en addr=0x%02X INT=%d Fs=%u Hz WHO_AM_I=0x%02X",
279 s_mpu_addr, APP_MPU_INT_GPIO, APP_IMU_SAMPLE_RATE_HZ, who);
280 ESP_RETURN_ON_ERROR(mpu_set_awake(false), TAG, "sleep inicial fallo");
281 return ESP_OK;
282 }
283
284 vTaskDelay(pdMS_TO_TICKS(150));
285 }
286
287 return err == ESP_OK ? ESP_FAIL : err;
288}

324esp_err_t hal_mpu6050_calibrate_gyro(mpu6050_calibration_t *cal)
325{
326 if (!cal) {
327 return ESP_ERR_INVALID_ARG;
328 }
329
330 memset(cal, 0, sizeof(*cal));
331 while (xSemaphoreTake(s_drdy_sem, 0) == pdTRUE) {
332 }
333 ESP_RETURN_ON_ERROR(mpu_set_awake(true), TAG, "wake calibrate fallo");
334 vTaskDelay(pdMS_TO_TICKS(50));
335 ESP_RETURN_ON_ERROR(mpu_write(REG_INT_ENABLE, MPU_DRDY_INT), TAG, "enable int fallo");
336
337 int64_t sum[3] = {0};
338 uint8_t buf[14];
339 uint16_t good = 0;
340
341 for (uint16_t i = 0; i < APP_GYRO_BIAS_SAMPLES; ++i) {
342 if (xSemaphoreTake(s_drdy_sem, pdMS_TO_TICKS(250)) != pdTRUE) {
343 continue;
344 }
345 if (mpu_read(REG_ACCEL_XOUT_H, buf, sizeof(buf)) != ESP_OK) {
346 continue;
347 }
348 sum[0] += be16(buf[8], buf[9]);
349 sum[1] += be16(buf[10], buf[11]);
350 sum[2] += be16(buf[12], buf[13]);
351 ++good;
352 }
353
354 (void)mpu_write(REG_INT_ENABLE, 0x00);
355 (void)mpu_set_awake(false);
356
357 if (good < (APP_GYRO_BIAS_SAMPLES / 2)) {
358 ESP_LOGW(TAG, "calibracion gyro insuficiente: %u/%u", good, APP_GYRO_BIAS_SAMPLES);
359 return ESP_ERR_TIMEOUT;
360 }
361
362 cal->gyro_bias_raw[0] = (int32_t)(sum[0] / good);
363 cal->gyro_bias_raw[1] = (int32_t)(sum[1] / good);
364 cal->gyro_bias_raw[2] = (int32_t)(sum[2] / good);
365 cal->bias_valid = true;
366
367 ESP_LOGI(TAG, "Bias gyro raw: [%ld, %ld, %ld]",
368 (long)cal->gyro_bias_raw[0], (long)cal->gyro_bias_raw[1], (long)cal->gyro_bias_raw[2]);
369 return ESP_OK;
370}

379esp_err_t hal_mpu6050_capture_window(const mpu6050_calibration_t *cal,
380 imu_sample_t *samples,
381 uint16_t target_samples,
382 uint16_t *captured_samples,
383 uint16_t *flags,
384 uint32_t t0_ms)
385{
386 if (!samples || !captured_samples || !flags) {
387 return ESP_ERR_INVALID_ARG;
388 }
389
390 *captured_samples = 0;
391 *flags = 0;
392
393 while (xSemaphoreTake(s_drdy_sem, 0) == pdTRUE) {
394 }
395 esp_err_t err = mpu_set_awake(true);
396 if (err != ESP_OK) {
397 ESP_LOGW(TAG, "wake capture fallo, intentando recovery...");
398 ESP_RETURN_ON_ERROR(hal_mpu6050_recover(), TAG, "recovery mpu fallo");
399 ESP_RETURN_ON_ERROR(mpu_set_awake(true), TAG, "wake capture tras recovery fallo");
400 }
401 vTaskDelay(pdMS_TO_TICKS(50));
402 ESP_RETURN_ON_ERROR(mpu_write(REG_INT_ENABLE, MPU_DRDY_INT), TAG, "int enable capture fallo");
403
404 uint8_t buf[14];
405 for (uint16_t i = 0; i < target_samples; ++i) {
406 if (xSemaphoreTake(s_drdy_sem, pdMS_TO_TICKS(250)) != pdTRUE) {
407 *flags |= APP_FLAG_IMU_SAMPLE_LOSS;
408 continue;
409 }
410
411 if (mpu_read(REG_ACCEL_XOUT_H, buf, sizeof(buf)) != ESP_OK) {
412 *flags |= APP_FLAG_IMU_SAMPLE_LOSS;
413 continue;
414 }
415
416 const int16_t ax_raw = be16(buf[0], buf[1]);
417 const int16_t ay_raw = be16(buf[2], buf[3]);
418 const int16_t az_raw = be16(buf[4], buf[5]);
419 int32_t gx_raw = be16(buf[8], buf[9]);
420 int32_t gy_raw = be16(buf[10], buf[11]);
421 int32_t gz_raw = be16(buf[12], buf[13]);
422
423 if (cal && cal->bias_valid) {
424 gx_raw -= cal->gyro_bias_raw[0];
425 gy_raw -= cal->gyro_bias_raw[1];
426 gz_raw -= cal->gyro_bias_raw[2];
427 *flags |= APP_FLAG_GYRO_BIAS_OK;
428 }
429
430 imu_sample_t *s = &samples[*captured_samples];
431 s->acc_mg[0] = (int16_t)lroundf(((float)ax_raw * 1000.0f) / APP_MPU_ACCEL_LSB_PER_G);
432 s->acc_mg[1] = (int16_t)lroundf(((float)ay_raw * 1000.0f) / APP_MPU_ACCEL_LSB_PER_G);
433 s->acc_mg[2] = (int16_t)lroundf(((float)az_raw * 1000.0f) / APP_MPU_ACCEL_LSB_PER_G);
434 s->gyro_dps10[0] = (int16_t)lroundf(((float)gx_raw * 10.0f) / APP_MPU_GYRO_LSB_PER_DPS);
435 s->gyro_dps10[1] = (int16_t)lroundf(((float)gy_raw * 10.0f) / APP_MPU_GYRO_LSB_PER_DPS);
436 s->gyro_dps10[2] = (int16_t)lroundf(((float)gz_raw * 10.0f) / APP_MPU_GYRO_LSB_PER_DPS);
437 s->t_rel_ms = (uint32_t)(esp_timer_get_time() / 1000ULL) - t0_ms;
438
439 ++(*captured_samples);
440 }
441
442 (void)mpu_write(REG_INT_ENABLE, 0x00);
443 (void)mpu_set_awake(false);
444
445 if (*captured_samples >= (target_samples * 80u / 100u)) {
446 *flags |= APP_FLAG_IMU_VALID_WINDOW;
447 }
448 if (*captured_samples == target_samples) {
449 *flags |= APP_FLAG_IMU_WINDOW_COMPLETE;
450 }
451
452 return ESP_OK;
453}

476 }
477
478 ESP_LOGW(TAG, "MPU recuperado. WHO_AM_I=0x%02X addr=0x%02X", who, s_mpu_addr);
