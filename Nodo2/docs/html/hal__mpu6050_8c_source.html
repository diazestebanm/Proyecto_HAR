77static esp_err_t mpu_set_awake(bool awake)
78{
79 if (awake) {
80 ESP_RETURN_ON_ERROR(mpu_write(REG_PWR_MGMT_1, MPU_CLKSEL_PLL_XGYRO), TAG, "despertar fallo");
81 ESP_RETURN_ON_ERROR(mpu_write(REG_PWR_MGMT_2, 0x00), TAG, "power mgmt2 fallo");
82 } else {
83 ESP_RETURN_ON_ERROR(mpu_write(REG_INT_ENABLE, 0x00), TAG, "deshabilitar int fallo");
84 ESP_RETURN_ON_ERROR(mpu_write(REG_PWR_MGMT_1, MPU_SLEEP_BIT), TAG, "sleep fallo");
85 }
86 return ESP_OK;
87}

93esp_err_t hal_mpu6050_init(TaskHandle_t imu_task_to_notify)
94{
95 s_imu_task = imu_task_to_notify;
96 ESP_RETURN_ON_ERROR(hal_i2c_init(), TAG, "I2C no disponible");
97
98 gpio_config_t io = {
99 .pin_bit_mask = (1ULL << APP_MPU_INT_GPIO),
100 .mode = GPIO_MODE_INPUT,
101 .pull_up_en = GPIO_PULLUP_DISABLE,
102 .pull_down_en = GPIO_PULLDOWN_DISABLE,
103 .intr_type = GPIO_INTR_POSEDGE,
104 };
105 ESP_RETURN_ON_ERROR(gpio_config(&io), TAG, "gpio int config fallo");
106 ESP_RETURN_ON_ERROR(gpio_install_isr_service(0), TAG, "isr service fallo");
107 ESP_RETURN_ON_ERROR(gpio_isr_handler_add(APP_MPU_INT_GPIO, mpu_isr_handler, NULL), TAG, "agregar isr fallo");
108
109 ESP_RETURN_ON_ERROR(mpu_set_awake(true), TAG, "despertar inicial fallo");
110 vTaskDelay(pdMS_TO_TICKS(100));
111
112 /*
113 * DLPF moderado y 25 Hz de salida. Con DLPF activado la base es 1 kHz,
114 * por eso: rate = 1000 / (1 + SMPLRT_DIV) => 25 Hz -> div = 39.
115 */
116 ESP_RETURN_ON_ERROR(mpu_write(REG_CONFIG, 0x03), TAG, "config dlpf fallo");
117 ESP_RETURN_ON_ERROR(mpu_write(REG_SMPLRT_DIV, 39), TAG, "smplrt_div fallo");
118 ESP_RETURN_ON_ERROR(mpu_write(REG_GYRO_CONFIG, 0x00), TAG, "gyro config fallo");
119 ESP_RETURN_ON_ERROR(mpu_write(REG_ACCEL_CONFIG, 0x00), TAG, "accel config fallo");
120 ESP_RETURN_ON_ERROR(mpu_write(REG_FIFO_EN, 0x00), TAG, "fifo en fallo");
121 ESP_RETURN_ON_ERROR(mpu_write(REG_INT_PIN_CFG, 0x00), TAG, "int pin cfg fallo");
122 ESP_RETURN_ON_ERROR(mpu_write(REG_INT_ENABLE, 0x00), TAG, "int disable fallo");
123
124 ESP_RETURN_ON_ERROR(mpu_set_awake(false), TAG, "sleep inicial fallo");
125 ESP_LOGI(TAG, "MPU6050 listo en INT=%d, Fs=%u Hz", APP_MPU_INT_GPIO, APP_IMU_SAMPLE_RATE_HZ);
126 return ESP_OK;
127}

131{
132 uint8_t who = 0;

141 ESP_LOGI(TAG, "WHO_AM_I=0x%02X", who);
142 return ESP_OK;

149esp_err_t hal_mpu6050_calibrate_gyro(mpu6050_calibration_t *cal)
150{
151 if (!cal) {
152 return ESP_ERR_INVALID_ARG;
153 }
154
155 memset(cal, 0, sizeof(*cal));
156 ESP_RETURN_ON_ERROR(mpu_set_awake(true), TAG, "wake calibrate fallo");
157 vTaskDelay(pdMS_TO_TICKS(50));
158 ESP_RETURN_ON_ERROR(mpu_write(REG_INT_ENABLE, MPU_DRDY_INT), TAG, "enable int fallo");
159
160 int64_t sum[3] = {0};
161 uint8_t buf[14];
162 uint16_t good = 0;
163
164 for (uint16_t i = 0; i < APP_GYRO_BIAS_SAMPLES; ++i) {
165 if (ulTaskNotifyTake(pdTRUE, pdMS_TO_TICKS(250)) == 0) {
166 continue;
167 }
168 if (mpu_read(REG_ACCEL_XOUT_H, buf, sizeof(buf)) != ESP_OK) {
169 continue;
170 }
171 sum[0] += be16(buf[8], buf[9]);
172 sum[1] += be16(buf[10], buf[11]);
173 sum[2] += be16(buf[12], buf[13]);
174 ++good;
175 }
176
177 mpu_write(REG_INT_ENABLE, 0x00);
178 mpu_set_awake(false);
179
180 if (good < (APP_GYRO_BIAS_SAMPLES / 2)) {
181 ESP_LOGW(TAG, "calibracion gyro insuficiente: %u/%u", good, APP_GYRO_BIAS_SAMPLES);
182 return ESP_ERR_TIMEOUT;
183 }
184
185 cal->gyro_bias_raw[0] = (int32_t)(sum[0] / good);
186 cal->gyro_bias_raw[1] = (int32_t)(sum[1] / good);
187 cal->gyro_bias_raw[2] = (int32_t)(sum[2] / good);
188 cal->bias_valid = true;
189
190 ESP_LOGI(TAG, "Bias gyro raw: [%ld, %ld, %ld]",
191 (long)cal->gyro_bias_raw[0], (long)cal->gyro_bias_raw[1], (long)cal->gyro_bias_raw[2]);
192 return ESP_OK;
193}

204esp_err_t hal_mpu6050_capture_window(const mpu6050_calibration_t *cal,
205 imu_sample_t *samples,
206 uint16_t target_samples,
207 uint16_t *captured_samples,
208 uint16_t *flags,
209 uint32_t t0_ms)
210{
211 if (!samples || !captured_samples || !flags) {
212 return ESP_ERR_INVALID_ARG;
213 }
214
215 *captured_samples = 0;
216 *flags = 0;
217
218 ESP_RETURN_ON_ERROR(mpu_set_awake(true), TAG, "wake capture fallo");
219 vTaskDelay(pdMS_TO_TICKS(50));
220 ESP_RETURN_ON_ERROR(mpu_write(REG_INT_ENABLE, MPU_DRDY_INT), TAG, "int enable capture fallo");
221
222 uint8_t buf[14];
223 for (uint16_t i = 0; i < target_samples; ++i) {
224 if (ulTaskNotifyTake(pdTRUE, pdMS_TO_TICKS(250)) == 0) {
225 *flags |= APP_FLAG_IMU_SAMPLE_LOSS;
226 continue;
227 }
228
229 if (mpu_read(REG_ACCEL_XOUT_H, buf, sizeof(buf)) != ESP_OK) {
230 *flags |= APP_FLAG_IMU_SAMPLE_LOSS;
231 continue;
232 }
233
234 const int16_t ax_raw = be16(buf[0], buf[1]);
235 const int16_t ay_raw = be16(buf[2], buf[3]);
236 const int16_t az_raw = be16(buf[4], buf[5]);
237 int32_t gx_raw = be16(buf[8], buf[9]);
238 int32_t gy_raw = be16(buf[10], buf[11]);
239 int32_t gz_raw = be16(buf[12], buf[13]);
240
241 if (cal && cal->bias_valid) {
242 gx_raw -= cal->gyro_bias_raw[0];
243 gy_raw -= cal->gyro_bias_raw[1];
244 gz_raw -= cal->gyro_bias_raw[2];
245 *flags |= APP_FLAG_GYRO_BIAS_OK;
246 }
247
248 imu_sample_t *s = &samples[*captured_samples];
249 s->acc_mg[0] = (int16_t)lroundf(((float)ax_raw * 1000.0f) / APP_MPU_ACCEL_LSB_PER_G);
250 s->acc_mg[1] = (int16_t)lroundf(((float)ay_raw * 1000.0f) / APP_MPU_ACCEL_LSB_PER_G);
251 s->acc_mg[2] = (int16_t)lroundf(((float)az_raw * 1000.0f) / APP_MPU_ACCEL_LSB_PER_G);
252 s->gyro_dps10[0] = (int16_t)lroundf(((float)gx_raw * 10.0f) / APP_MPU_GYRO_LSB_PER_DPS);
253 s->gyro_dps10[1] = (int16_t)lroundf(((float)gy_raw * 10.0f) / APP_MPU_GYRO_LSB_PER_DPS);
254 s->gyro_dps10[2] = (int16_t)lroundf(((float)gz_raw * 10.0f) / APP_MPU_GYRO_LSB_PER_DPS);
255 s->t_rel_ms = (uint32_t)(esp_timer_get_time() / 1000ULL) - t0_ms;
256
257 ++(*captured_samples);
258 }
259
260 mpu_write(REG_INT_ENABLE, 0x00);
261 mpu_set_awake(false);
262
263 if (*captured_samples >= (target_samples * 80u / 100u)) {
264 *flags |= APP_FLAG_IMU_VALID_WINDOW;
265 }
266 if (*captured_samples == target_samples) {
267 *flags |= APP_FLAG_IMU_WINDOW_COMPLETE;
268 }
269
270 return ESP_OK;
271}
